def fetch_r(key)
  if self.has_key?(key) and not self[key].is_a?(Hash)
    return self[key]
  else
    self.each do |val|
	value = false
	if val.is_a?(Array)
        val.each do |x|
          value = x.fetch_r(key) if x.is_a?(Hash)
	    return value if value

end

    end
  end
