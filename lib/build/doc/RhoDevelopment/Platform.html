def self.terminate_process(pid)
  if self.windows?
    system "taskkill /F /PID #{pid}"
  elsif self.osx?
    Process.kill('SIGTERM', pid)
  else
    raise "Unknown platform #{RbConfig::CONFIG['host_os']}"
  end
end
